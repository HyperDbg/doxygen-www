428
429 //
430 // We are in kernel, so we should read MSR GS_BASE
431 // IA32_GS_BASE 0xC0000101
432 // IA32_KERNEL_GS_BASE 0xC0000102
433 // IA32_KERNEL_GS_BASE is currently user's gs as
434 // we are in the kernel-mode, if we need to intercept

437 //

476 //

478 //

480 // read/write on the gs:188 as we intercept it on
481 // hypervisor side by exception bitmap on #DBs
