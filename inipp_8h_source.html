70 template <class CharT>
71 inline bool
72 replace(std::basic_string<CharT> & str, const std::basic_string<CharT> & from, const std::basic_string<CharT> & to)
73 {
74  auto changed = false;
75  size_t start_pos = 0;
76  while ((start_pos = str.find(from, start_pos)) != std::basic_string<CharT>::npos)
77  {
78  str.replace(start_pos, from.length(), to);
79  start_pos += to.length();
80  changed = true;
81  }
82  return changed;
83 }

89 extract(const std::basic_string<CharT> & value, T & dst)
90 {
91  CharT c;
92  std::basic_istringstream<CharT> is {value};
93  T result;
94  if ((is >> std::boolalpha >> result) && !(is >> c))
95  {
96  dst = result;
97  return true;
98  }
99  else
100  {
101  return false;

103 }

201  void parse(std::basic_istream<CharT> & is)
202  {
203  String line;
204  String section;
205  const std::locale loc {"C"};
206  while (std::getline(is, line))
207  {
208  detail::ltrim(line, loc);
209  detail::rtrim(line, loc);
210  const auto length = line.length();
211  if (length > 0)
212  {
213  const auto pos = std::find_if(line.begin(), line.end(), [this](CharT ch) { return format->is_assign(ch); });
214  const auto & front = line.front();
215  if (format->is_comment(front))
216  {
217  }
218  else if (format->is_section_start(front))
219  {
220  if (format->is_section_end(line.back()))
221  section = line.substr(1, length - 2);
222  else
223  errors.push_back(line);
224  }
225  else if (pos != line.begin() && pos != line.end())
226  {
227  String variable(line.begin(), pos);
228  String value(pos + 1, line.end());
229  detail::rtrim(variable, loc);
230  detail::ltrim(value, loc);
231  auto & sec = sections[section];
232  if (sec.find(variable) == sec.end())
233  sec.insert(std::make_pair(variable, value));
234  else
235  errors.push_back(line);
236  }
237  else
238  {
239  errors.push_back(line);
240  }
241  }
242  }
243  }

245  void interpolate()
246  {
247  int global_iteration = 0;
248  auto changed = false;
249  // replace each "${variable}" by "${section:variable}"
250  for (auto & sec : sections)
251  replace_symbols(local_symbols(sec.first, sec.second), sec.second);
252  // replace each "${section:variable}" by its value
253  do
254  {
255  changed = false;
256  const auto syms = global_symbols();
257  for (auto & sec : sections)
258  changed |= replace_symbols(syms, sec.second);
259  } while (changed && (max_interpolation_depth > global_iteration++));
260  }

289  const Symbols local_symbols(const String & sec_name, const Section & sec) const
290  {
291  Symbols result;
292  for (const auto & val : sec)
293  result.push_back(std::make_pair(format->local_symbol(val.first), format->global_symbol(sec_name, val.first)));
294  return result;

297  const Symbols global_symbols() const
298  {
299  Symbols result;
300  for (const auto & sec : sections)
301  for (const auto & val : sec.second)
302  result.push_back(
303  std::make_pair(format->global_symbol(sec.first, val.first), val.second));
304  return result;
305  }
306 
307  bool replace_symbols(const Symbols & syms, Section & sec) const
308  {
309  auto changed = false;
310  for (auto & sym : syms)
311  for (auto & val : sec)
312  changed |= detail::replace(val.second, sym.first, sym.second);
313  return changed;
314  }
